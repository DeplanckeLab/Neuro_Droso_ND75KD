## data:  boxplot_data$AUC_OXPHOS_68_Porcelli[boxplot_data$Neuron_category == "Optic lobe"] and boxplot_data$AUC_OXPHOS_68_Porcelli[boxplot_data$Neuron_category == "Glia"]
## W = 326033, p-value = 0.8875
## alternative hypothesis: true location shift is not equal to 0
# Extracting the relevant metadata
boxplot_data <- as.data.frame(data.seurat@meta.data)

# Perform pairwise Wilcoxon tests and annotate p-values
p <- ggplot(boxplot_data, aes(x = Neuron_category, y = AUC_OXPHOS_68_Porcelli, fill = Neuron_category)) +
  geom_boxplot() +
  labs(x = NULL, y = "AUC_OXPHOS_68_Porcelli") +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 90, vjust = 0.5, hjust = 1)) +
  scale_fill_manual(values = steinPalette) +
  NoLegend() +
  stat_compare_means(method = "wilcox.test", label = "p.signif", 
                     comparisons = list(
                       c("Central Brain A", "Central Brain B"),
                       c("Central Brain A", "Optic lobe"),
                       c("Central Brain A", "Glia"),
                       c("Central Brain B", "Optic lobe"),
                       c("Central Brain B", "Glia"),
                       c("Optic lobe", "Glia")
                     ))

p
